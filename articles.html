this is just simple function to compile cpp from command line

#usage
$compile filename

function compile (){
      local objname="${1}"
      local srcname="${1}.cpp"
      clang++ -o "$objname" "$srcname"
      ;./"$objname"
      echo ">- warning -<"
      rm -i "$objname"
 }
